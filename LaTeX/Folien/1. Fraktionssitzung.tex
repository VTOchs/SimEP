\newcommand{\partyseats}[1]{%
    \DTLforeach{caucus}{\party=party,\total=total}{%

}

    \DTLforeach{caucus}{\party=party,\countries=countries}{%
        \ifthenelse{\equal{#1}{\party}}{\countries}{}%
